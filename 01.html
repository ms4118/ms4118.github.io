# Function to calculate m (required storage locations) def calculate_required_m(*args): # Get the selected policy (dedicated or random) policy = Element("storage_policy").value if policy == "dedicated": # DEDICATED STORAGE: Sum of each product's maximum inventory # m_d = Σ max(I_j) for j=1 to n max_inventory_list = [] n = int(Element("product_count").value) for i in range(n): # Get the value for each product's max inventory try: max_val = float(Element(f"max_inv_{i}").value) max_inventory_list.append(max_val) except: # If input is invalid, use 0 max_inventory_list.append(0) # Calculate m for dedicated storage m = sum(max_inventory_list) # Create the formula text to show the user formula_text = "m = Σ max(I_j) = " + " + ".join([str(int(x)) for x in max_inventory_list]) + f" = {int(m)}" else: # RANDOM STORAGE: Maximum of the total inventory at any time # m_r = max(Σ I_j(t)) for all time periods t time_period_sums = [] # Get the total inventory for each time period (assuming 3 periods for this example) for t in range(3): try: sum_val = float(Element(f"time_sum_{t}").value) time_period_sums.append(sum_val) except: time_period_sums.append(0) # Calculate m for random storage (find the maximum sum) m = max(time_period_sums) # Create the formula text to show the user formula_text = f"m = max(Σ I_j(t)) = max({time_period_sums}) = {int(m)}" # Display the result to the user result_html = f"""
Result: m = {int(m)} storage locations required.

Formula: {formula_text}

Policy: {'Dedicated' if policy == 'dedicated' else 'Random'} Storage

""" Element("m_result").write(result_html) # Connect the calculate button to the function - FIXED LINE Element("calculate_m").add_event_listener("click", calculate_required_m)
